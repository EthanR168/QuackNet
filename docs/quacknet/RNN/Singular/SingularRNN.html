 22        Args:
 23            hiddenStateActivationFunction (str): Name of activation function for hidden states (e.g., "relu", "sigmoid").
 24            outputLayerActivationFunction (str): Name of activation function for output layer.
 25            lossFunction (str): Name of the loss function to use (e.g., "mse", "mae", "cross entropy").
 26            useBatches (bool, optional): Whether to use batching during training. Default is False.
 27            batchSize (int, optional): Batch size if batching is enabled. Default is 64.

 30        self.hiddenWeight = None
 31        self.bias = None
 32        self.outputWeight = None
 33        self.outputBias = None
 34        self.hiddenState = None
 35

 37            "relu": relu,
 38            "sigmoid": sigmoid,
 39            "linear": linear,
 40            "tanh": tanH,
 41            "softmax": softMax,
 42        }
 43        if(hiddenStateActivationFunction.lower() not in funcs):
 44            raise ValueError(f"Activation function not made: {hiddenStateActivationFunction.lower()}")
 45        if(outputLayerActivationFunction.lower() not in funcs):
 46            raise ValueError(f"Activation function not made: {outputLayerActivationFunction.lower()}")
 47        self.hiddenStateActivationFunction = funcs[hiddenStateActivationFunction.lower()]
 48        self.outputLayerActivationFunction = funcs[outputLayerActivationFunction.lower()]
 49

 51            relu: ReLUDerivative,
 52            sigmoid: SigmoidDerivative,
 53            linear: LinearDerivative,
 54            tanH: TanHDerivative,
 55            softMax: SoftMaxDerivative,
 56        }
 57        self.activationDerivative = derivs[self.hiddenStateActivationFunction]
 58        self.outputLayerDerivative = derivs[self.outputLayerActivationFunction]
 59
 60        lossFunctionDict = {
 61            "mse": MSELossFunction,
 62            "mae": MAELossFunction,
 63            "cross entropy": CrossEntropyLossFunction,"cross": CrossEntropyLossFunction,
 64        }
 65        self.lossFunction = lossFunctionDict[lossFunction.lower()]
 66        lossDerivs = {
 67            MSELossFunction: MSEDerivative,
 68            MAELossFunction: MAEDerivative,
 69            CrossEntropyLossFunction: CrossEntropyLossDerivative,
 70        }
 71        self.lossDerivative = lossDerivs[self.lossFunction]
 72
 73        self.useBatches = useBatches
 74        self.batchSize = batchSize

 78    def forwardSequence(self, inputData): # goes through the whole sequence / time steps
 79        batchSize, sequenceLegth, _ = inputData.shape
 80        
 81        preActivations = []
 82        allHiddenStates = []
 83
 84        for i in range(sequenceLegth):
 85            xi = inputData[:, i, :].reshape(batchSize, -1, 1)
 86            preAct, outputPreAct, output = self._oneStep(xi)
 87            preActivations.append(preAct)
 88            allHiddenStates.append(self.hiddenState.copy())
 89        self.preActivations = preActivations
 90        self.allHiddenStates = allHiddenStates
 91        self.outputPreAct = outputPreAct
 92        return output
 93
 94    def _oneStep(self, inputData): # forward prop on 1 time step
 95        preActivation, self.hiddenState = self._calculateHiddenLayer(inputData, self.hiddenState, self.inputWeight, self.hiddenWeight, self.bias, self.hiddenStateActivationFunction)
 96        preAct, output = self._calculateOutputLayer(self.hiddenState, self.outputWeight, self.outputBias, self.outputLayerActivationFunction)
 97        return preActivation, preAct, output

 99    def _calculateHiddenLayer(self, inputData, lastHiddenState, inputWeight, hiddenWeight, bias, activationFunction): # a( w_x * x + w_h * h + b )
100        weighttedInp = np.matmul(inputWeight, inputData)
101        weightedHidden = np.matmul(hiddenWeight, lastHiddenState)
102        biasBrodcast = bias.reshape(1, -1, 1)
103        preActivation = weighttedInp + weightedHidden + biasBrodcast
104        newHiddenState = activationFunction(preActivation)
105        return preActivation, newHiddenState
106
107    def _calculateOutputLayer(self, input, outputWeight, outputBias, activationFunction): # a( w_o * o + b_o)
108        weigthedOutput = np.matmul(outputWeight, input)
109        outputBiasBrodcast = outputBias.reshape(1, -1, 1)
110        preActivation = weigthedOutput + outputBiasBrodcast
111        output = activationFunction(preActivation)
112        return preActivation, output

114    def _initialiseWeights(self, outputSize, inputSize, activationFunction):
115        if(activationFunction == relu):
116            bounds = math.sqrt(2 / inputSize) # He initialisation
117        elif(activationFunction == sigmoid):
118            bounds = math.sqrt(6 / (inputSize + outputSize)) # Xavier initialisation
119        else:
120            bounds = 1 / np.sqrt(inputSize) # default
121        w = np.random.normal(0, bounds, size=(outputSize, inputSize))
122        return w

124    def initialiseWeights(self, inputSize, hiddenSize, outputSize):
125        """
126        Initializes weights and biases for input, hidden, and output layers,

135        self.hiddenWeight = self._initialiseWeights(hiddenSize, hiddenSize, self.hiddenStateActivationFunction)
136        self.outputWeight = self._initialiseWeights(outputSize, hiddenSize, self.outputLayerActivationFunction)
137        self.bias = np.zeros((hiddenSize, 1))
138        self.outputBias = np.zeros((outputSize, 1))
139        self.hiddenState = np.zeros((hiddenSize, 1))
140        self.inputSize = inputSize
141        self.hiddenSize = hiddenSize
142        self.outputSize = outputSize

172    def train(self, inputData, labels, alpha = 0.001, beta1 = 0.9, beta2 = 0.999, epsilon = 1e-8):
173        """
174        Trains the model on the given input sequences and labels using the Adam optimizer,
175        then calculates and returns the loss.
176
177        Args:
178            inputData (ndarray): 3D array of input data with shape (batchSize, sequenceLength, inputSize).
179            labels (ndarray): True labels corresponding to the input data.
180            alpha (float, optional): Learning rate for the optimizer. Default is 0.001.
181            beta1 (float, optional): Beta1 parameter for Adam optimizer. Default is 0.9.
182            beta2 (float, optional): Beta2 parameter for Adam optimizer. Default is 0.999.
183            epsilon (float, optional): Epsilon parameter for Adam optimizer to prevent division by zero. Default is 1e-8.
184
185        Returns:
186            float: Calculated loss between model outputs and true labels.
187        """
188        assert np.array(inputData).ndim == 3, f"Dimension wrong size, got {np.array(inputData).ndim}, expected 3"
189        AllOutputs = self.optimiser(inputData, labels, alpha, beta1, beta2, epsilon)
190        AllOutputs = np.reshape(AllOutputs, (np.array(labels).shape))
191        loss = self.lossFunction(AllOutputs, labels)
192        return loss
